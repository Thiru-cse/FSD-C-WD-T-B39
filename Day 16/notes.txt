    console.log("Button Clicked");
}

MouseOver:
var div = document.createElement("div");
div.innerText = "Click";
document.body.append(div);

div.addEventListener("mouseover", () => div.style.background = "red");

Confirm:
var button = document.createElement("button");
button.setAttribute("type", "button");
button.innerText = "Confirm";
document.body.append(button);

button.addEventListener("click", dabba);

function dabba() {
    var res = confirm("You are not a robot");
    console.log(res);
}

Example: Append vs AppendChild
var div1 = document.createElement("div");
div1.setAttribute("id", "div1");
div1.innerText = "Hello";

var div2 = document.createElement("div");
div2.setAttribute("id", "div2");
div2.innerText = "World";

document.body.appendChild(div1, div2);
